位元運算<br>
可以，這很星爆<br>

```cpp
#include <iostream>
using namespace std;

int n,limit,cnt;
int x[21],k=1;
//行，左對角線，右對角線
void DFS(int row,int left,int right)
{
    if(row!=limit)
    {
        int pos=limit&~(row|left|right);
        while(pos)
        {
            //找到的可以放皇后的位置
            int p=pos&-pos;// pos & (~pos+1);
            pos&=pos-1;
            DFS(row|p,(left|p)<<1,(right|p)>>1);
        }
    }
    else
    {
        cnt++;
    }
}

int main()
{
    while(cin>>n)
    {
        cnt=0;
        limit=(1<<n)-1;
        DFS(0,0,0);
        cout<<cnt<<endl;
    }
}
```
